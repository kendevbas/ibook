src/
├── js/
│   ├── main.js                # Основной файл, где происходит инициализация всего приложения
│   ├── utils.js               # Утилиты и вспомогательные функции
│   ├── classes/
│   │   ├── User.js            # Класс пользователя, включая роли и доступ
│   │   ├── Product.js         # Класс продукта (книга)
│   │   ├── Order.js           # Класс заказа
│   │   └── Cart.js            # Класс корзины для покупателя
│   ├── modules/
│   │   ├── api.js             # Функции для работы с API
│   │   ├── format.js          # Функции для форматирования данных (цены, даты)
│   │   ├── auth.js            # Функции для аутентификации и авторизации
│   │   └── storage.js         # Работа с локальным хранилищем (например, сохранение сессий)
│   ├── admin/                 # Функции и классы, специфичные для админской панели
│   │   ├── dashboard.js       # Основной скрипт панели администратора
│   │   ├── manageProducts.js  # Управление товарами (добавление, редактирование, удаление)
│   │   └── manageOrders.js    # Управление заказами (просмотр, обработка)
│   └── customer/              # Функции и классы для покупательской части сайта
│       ├── productList.js     # Отображение списка книг для покупателя
│       ├── cartView.js        # Управление отображением и логикой корзины
│       └── checkout.js        # Оформление заказа
├── styles/
│   ├── core.scss              # Основные стили проекта
│   ├── admin.scss             # Стили для админской панели
│   └── customer.scss          # Стили для покупательской части
├── assets/                    # Статические файлы (изображения, иконки)
│   ├── images/
│   ├── fonts/
│   └── icons/
└── templates/
    ├── admin/                 # Шаблоны HTML для админской панели
    │   ├── dashboard.html
    │   ├── manageProducts.html
    │   └── manageOrders.html
    └── customer/              # Шаблоны HTML для покупательской части
        ├── productList.html
        ├── cart.html
        └── checkout.html
